    Deployment Troubleshooting Guide¶

    This comprehensive troubleshooting guide covers common issues encountered when deploying and operating the Crew Appraisals System across different environments and deployment methods.

    Quick Diagnostic Commands¶

    Health Check Commands¶

    # Quick system health check
curl -f http://localhost:8000/api/v1/health || echo "API unhealthy"
curl -f http://localhost:8000/api/v1/health/db || echo "Database unhealthy"
curl -f http://localhost:8000/api/v1/health/redis || echo "Redis unhealthy"

# Docker health checks
docker compose ps
docker compose logs --tail=50 backend postgres redis

# Kubernetes health checks
kubectl get pods -l app=crew-appraisals
kubectl describe pod <pod-name>
kubectl logs <pod-name> --tail=50

    Database Connection Issues¶

    Symptoms¶

    • Connection refused errors
    • SQLSTATE[08006] connection failure
    • Slow database operations
    • Connection timeout errors

    Docker Compose Troubleshooting¶

    Issue: PostgreSQL not starting¶

    # Check PostgreSQL logs
docker compose logs postgres

# Common error: "database system was shut down"
# Solution: Remove postgres data volume and restart
docker compose down
docker volume rm $(docker compose config --volumes | grep postgres)
docker compose up -d postgres

# Wait for PostgreSQL to initialize
docker compose logs -f postgres

    Issue: Connection refused to PostgreSQL¶

    # Verify PostgreSQL is running and healthy
docker compose ps postgres
docker compose exec postgres pg_isready -U crewing_user -d crew_appraisals

# Check network connectivity between services
docker compose exec backend ping postgres

# Verify environment variables
docker compose exec backend env | grep -E "(DB_|DATABASE_)"

    Issue: Authentication failed¶

    # Check PostgreSQL user and database
docker compose exec postgres psql -U postgres -c "\\du"
docker compose exec postgres psql -U postgres -c "\\l"

# Reset password if needed
docker compose exec postgres psql -U postgres -c "ALTER USER crewing_user PASSWORD 'new_password';"

# Update .env file with correct credentials
vim docker/.env

    Database Performance Issues¶

    Issue: Slow queries¶

    # Enable query logging (development only)
docker compose exec postgres psql -U postgres -d crew_appraisals -c "
ALTER SYSTEM SET log_statement = 'all';
ALTER SYSTEM SET log_min_duration_statement = 100;
SELECT pg_reload_conf();
"

# Monitor slow queries
docker compose logs -f postgres | grep "duration:"

# Check active connections
docker compose exec postgres psql -U postgres -c "
SELECT pid, usename, application_name, client_addr, state, query
FROM pg_stat_activity
WHERE state = 'active';"

    Issue: Connection pool exhaustion¶

    # Check connection pool settings in backend code
# src/core/database.py
engine = create_async_engine(
    DATABASE_URL,
    pool_size=20,        # Increase if needed
    max_overflow=10,     # Allow burst connections
    pool_pre_ping=True,  # Validate connections
    pool_recycle=3600,   # Recycle connections hourly
)

    Kubernetes Database Issues¶

    Issue: PostgreSQL pod in pending state¶

    # Check resource availability
kubectl describe node
kubectl get pv,pvc

# Check storage class
kubectl get storageclass

# Create PV manually if needed
kubectl apply -f - <<EOF
apiVersion: v1
kind: PersistentVolume
metadata:
  name: postgres-pv
spec:
  capacity:
    storage: 20Gi
  accessModes:
    - ReadWriteOnce
  hostPath:
    path: /data/postgres
EOF

    Issue: Database initialization fails in Kubernetes¶

    # Check init container logs
kubectl logs <postgres-pod> -c init-db

# Check permissions
kubectl exec <postgres-pod> -- ls -la /var/lib/postgresql/data

# Fix ownership if needed
kubectl exec <postgres-pod> -- chown -R postgres:postgres /var/lib/postgresql/data

    Port Conflicts¶

    Symptoms¶

    • Port already in use errors
    • Services failing to start
    • Connection refused on expected ports

    Docker Port Conflicts¶

    Issue: Port 8000 already in use¶

    # Find process using port 8000
sudo netstat -tulpn | grep :8000
# or
sudo lsof -i :8000

# Kill conflicting process
sudo kill -9 <PID>

# Use alternative port
export API_PORT=8001
docker compose up -d

    Issue: PostgreSQL port 5432 in use¶

    # Check if local PostgreSQL is running
sudo service postgresql status

# Stop local PostgreSQL
sudo service postgresql stop

# Or use different port
export DB_PORT=5433
docker compose up -d

    Port Mapping Solutions¶

    # docker-compose.override.yaml
services:
  backend:
    ports:
      - "8001:8000"  # Map to different host port

  postgres:
    ports:
      - "5433:5432"  # Use different PostgreSQL port

  redis:
    ports:
      - "6380:6379"  # Use different Redis port

    Kubernetes Port Issues¶

    Issue: Service not accessible¶

    # Check service endpoints
kubectl get svc,ep

# Check if pods are ready
kubectl get pods -o wide

# Test service connectivity internally
kubectl run test-pod --rm -i --tty --image=busybox -- sh
/ # nslookup crew-appraisals-service
/ # wget -qO- crew-appraisals-service/api/v1/health

    Issue: Ingress not routing traffic¶

    # Check ingress controller
kubectl get pods -n ingress-nginx

# Check ingress configuration
kubectl describe ingress crew-appraisals

# Verify DNS resolution
nslookup your-domain.com

# Check SSL certificate
kubectl describe certificate crew-appraisals-tls

    Docker Issues¶

    Container Startup Problems¶

    Issue: Container exits immediately¶

    # Check exit code and logs
docker compose ps
docker compose logs backend

# Run container interactively for debugging
docker compose run --rm backend bash

# Check if entrypoint script is executable
docker compose exec backend ls -la /entrypoint.sh
docker compose exec backend chmod +x /entrypoint.sh

    Issue: Python import errors¶

    # Check Python path and installed packages
docker compose exec backend python -c "import sys; print(sys.path)"
docker compose exec backend pip list

# Reinstall dependencies if needed
docker compose exec backend poetry install

# Check for missing system dependencies
docker compose exec backend apt list --installed | grep -E "(gcc|python|postgresql)"

    Build Issues¶

    Issue: Docker build fails¶

    # Build with verbose output
docker compose build --progress=plain --no-cache backend

# Check disk space
df -h

# Clean up Docker space
docker system prune -a
docker volume prune

    Issue: Poetry installation fails¶

    # Add debug information to Dockerfile
RUN curl -sSL https://install.python-poetry.org | python3 - \
    && ln -s /opt/poetry/bin/poetry /usr/local/bin/poetry \
    && poetry --version  # Verify installation

    Memory and Resource Issues¶

    Issue: Out of memory errors¶

    # Check container memory usage
docker stats

# Increase Docker memory limit (Docker Desktop)
# Settings > Resources > Memory > Increase to 8GB

# Add memory limits to compose file
services:
  backend:
    deploy:
      resources:
        limits:
          memory: 2g
        reservations:
          memory: 1g

    Issue: Disk space issues¶

    # Check disk usage
docker system df

# Clean up unused resources
docker system prune -a --volumes
docker volume prune

# Remove unused images
docker image prune -a

    Performance Problems¶

    High Response Times¶

    Issue: Slow API responses¶

    # Check system resources
docker stats
htop  # or top

# Enable application performance monitoring
export LOG_LEVEL=DEBUG
docker compose restart backend

# Check database performance
docker compose exec postgres psql -U crewing_user -d crew_appraisals -c "
SELECT query, calls, mean_exec_time, total_exec_time
FROM pg_stat_statements
ORDER BY total_exec_time DESC
LIMIT 10;"

    Issue: Memory leaks¶

    # Monitor memory usage over time
while true; do
    echo "$(date): $(docker stats --no-stream --format 'table {{.Name}}\t{{.MemUsage}}' | grep backend)"
    sleep 60
done

# Check for memory leaks in application
docker compose exec backend python -c "
import psutil
process = psutil.Process()
print(f'Memory: {process.memory_info().rss / 1024 / 1024:.1f} MB')
print(f'Open files: {len(process.open_files())}')
print(f'Connections: {len(process.connections())}')
"

    Database Performance¶

    Issue: Slow database queries¶

    -- Check slow queries
SELECT query, calls, total_exec_time, mean_exec_time, rows
FROM pg_stat_statements
WHERE mean_exec_time > 100
ORDER BY mean_exec_time DESC;

-- Check missing indexes
SELECT schemaname, tablename, attname, n_distinct, correlation
FROM pg_stats
WHERE schemaname = 'public'
  AND n_distinct > 100
  AND correlation < 0.1;

-- Check table sizes
SELECT schemaname, tablename,
       pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) as size
FROM pg_tables
WHERE schemaname = 'public'
ORDER BY pg_total_relation_size(schemaname||'.'||tablename) DESC;

    SSL/TLS Certificate Issues¶

    Let's Encrypt Certificate Problems¶

    Issue: Certificate generation fails¶

    # Check cert-manager logs (Kubernetes)
kubectl logs -n cert-manager -l app=cert-manager

# Verify DNS configuration
dig your-domain.com
dig _acme-challenge.your-domain.com TXT

# Check rate limits
curl -s "https://crt.sh/?q=%.your-domain.com&output=json" | jq length

# Manual certificate generation (Docker)
docker run -it --rm \
    -v /path/to/ssl:/etc/letsencrypt \
    certbot/certbot certonly \
    --manual \
    --preferred-challenges dns \
    -d your-domain.com

    Issue: Certificate renewal fails¶

    # Test certificate renewal
certbot renew --dry-run

# Check certificate expiry
openssl x509 -in /path/to/cert.pem -text -noout | grep "Not After"

# Force renewal
certbot renew --force-renewal

    Self-Signed Certificate Issues¶

    Issue: Browser security warnings¶

    # Generate proper self-signed certificate
openssl req -x509 -newkey rsa:4096 -keyout key.pem -out cert.pem -days 365 -nodes \
    -subj "/C=US/ST=State/L=City/O=Organization/CN=localhost" \
    -addext "subjectAltName=DNS:localhost,DNS:*.localhost,IP:127.0.0.1"

# Trust certificate in browser or OS
# Chrome: chrome://settings/certificates
# Firefox: about:config > security.tls.insecure_fallback_hosts

    Monitoring and Logging Issues¶

    Prometheus/Grafana Problems¶

    Issue: Metrics not appearing¶

    # Check if metrics endpoint is accessible
curl http://localhost:8000/api/v1/metrics

# Verify Prometheus configuration
docker compose exec prometheus cat /etc/prometheus/prometheus.yml

# Check Prometheus targets
curl http://localhost:9090/api/v1/targets

# Restart Prometheus with new config
docker compose restart prometheus

    Issue: Grafana dashboard not loading¶

    # Check Grafana logs
docker compose logs grafana

# Verify datasource configuration
curl -u admin:admin http://localhost:3001/api/datasources

# Import dashboard manually
curl -X POST \
    -H "Content-Type: application/json" \
    -d @dashboard.json \
    -u admin:admin \
    http://localhost:3001/api/dashboards/db

    Issue: Fluentd not collecting logs¶

docker compose exec fluentd cat /fluentd/etc/fluent.conf

# Test log forwarding
docker compose exec backend logger "Test log message"

# Check Fluentd buffer
docker compose exec fluentd ls -la /var/log/fluentd/

# Restart Fluentd
docker compose restart fluentd

    Network Connectivity Issues¶

    Service Discovery Problems¶

    Issue: Services cannot communicate¶

    # Check network configuration
docker network ls
docker network inspect $(docker compose config --format json | jq -r '.networks | keys[]')

# Test connectivity between services
docker compose exec backend ping postgres
docker compose exec backend ping redis

# Check service names in DNS
docker compose exec backend nslookup postgres
docker compose exec backend nslookup redis

    Firewall Issues¶

    Issue: External access blocked¶

    # Check iptables rules
sudo iptables -L -n

# Check ufw status (Ubuntu)
sudo ufw status

# Allow specific ports
sudo ufw allow 8000
sudo ufw allow 443

# Check if services are binding to correct interfaces
netstat -tulpn | grep -E ":(8000|5432|6379)"

    Kubernetes Network Issues¶

    Issue: Pod-to-pod communication fails¶

    # Check network policies
kubectl get networkpolicies

# Test connectivity between pods
kubectl exec <pod1> -- ping <pod2-ip>

# Check service mesh configuration (if using Istio)
kubectl get virtualservices,destinationrules

    Environment-Specific Issues¶

    Development Environment¶

    Issue: Hot reloading not working¶

    # Check volume mounts
docker compose config | grep -A 5 volumes

# Verify file watcher limits (Linux)
echo fs.inotify.max_user_watches=524288 | sudo tee -a /etc/sysctl.conf
sudo sysctl -p

# Use polling for file changes
export CHOKIDAR_USEPOLLING=true

    Production Environment¶

    Issue: Resource constraints¶

    # Check system resources
free -h

iostat -x 1 5

# Monitor container resource usage
docker stats --no-stream

# Check swap usage
cat /proc/swaps
swapon --show

    Issue: Security scanning alerts¶

    # Scan for vulnerabilities
docker run --rm -v /var/run/docker.sock:/var/run/docker.sock \
    aquasec/trivy image crew-appraisals/backend:latest

# Update base images
docker pull python:3.11-slim
docker compose build --no-cache backend

# Check for security updates
docker compose exec backend apt list --upgradable

    Emergency Procedures¶

    Complete System Recovery¶

    Scenario: Total system failure¶

    # 1. Stop all services
docker compose down

# 2. Backup any salvageable data
docker run --rm -v postgres_data:/data -v $(pwd):/backup \
    busybox tar czf /backup/postgres_backup.tar.gz /data

# 3. Reset environment
docker system prune -a --volumes
docker compose pull
docker compose build --no-cache

# 4. Restore from backup
docker run --rm -v postgres_data:/data -v $(pwd):/backup \
    busybox tar xzf /backup/postgres_backup.tar.gz -C /

# 5. Start services
docker compose up -d

# 6. Verify system health
./scripts/health-check.sh

    Database Recovery¶

    Scenario: Database corruption¶

    # 1. Stop application
docker compose stop backend

# 2. Backup current state
docker compose exec postgres pg_dump -U crewing_user crew_appraisals > corrupt_backup.sql

# 3. Check database integrity
docker compose exec postgres pg_dump -U crewing_user --schema-only crew_appraisals > schema.sql

# 4. Restore from last known good backup
cat last_good_backup.sql | docker compose exec -T postgres psql -U crewing_user crew_appraisals

# 5. Restart application
docker compose start backend

    Monitoring and Alerting Setup¶

    Health Check Automation¶

    #!/bin/bash
# scripts/health-monitor.sh

SERVICES=("http://localhost:8000/api/v1/health" "http://localhost:8000/api/v1/health/db")
ALERT_EMAIL="admin@example.com"

for service in "${SERVICES[@]}"; do
    if ! curl -f -s "$service" >/dev/null; then
        echo "ALERT: $service is down" | mail -s "Service Alert" "$ALERT_EMAIL"
    fi
done

    Log Analysis Scripts¶

    #!/bin/bash
# scripts/analyze-logs.sh

# Extract error patterns
docker compose logs backend | grep -E "(ERROR|CRITICAL)" | tail -100

# Count error types
docker compose logs backend | grep ERROR | awk '{print $NF}' | sort | uniq -c

# Monitor response times
docker compose logs nginx | grep -o 'request_time=[0-9.]*' | cut -d= -f2 | sort -n

    Preventive Measures¶

    Regular Maintenance¶

    #!/bin/bash
# scripts/maintenance.sh

# Update Docker images
docker compose pull

# Clean up unused resources
docker system prune -f

# Backup databases
pg_dump crew_appraisals > backup_$(date +%Y%m%d).sql

# Check disk space
df -h | grep -E "(8[0-9]|9[0-9])%" && echo "ALERT: Disk space low"

# Rotate logs
find /var/log -name "*.log" -mtime +30 -delete

    Monitoring Checklist¶

    • API response times < 200ms
    • Database connection pool < 80% utilization
    • Disk space usage < 80%
    • Memory usage < 80%
    • CPU usage < 70% average
    • SSL certificates valid for > 30 days
    • All services healthy and responding
    • Error rate < 1%
    • Backup jobs completing successfully

    Remember: When troubleshooting, start with the simplest explanations and work your way up to more complex scenarios. Always check logs first, verify configuration second, and test connectivity third.
