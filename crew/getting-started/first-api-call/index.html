    First API Call Tutorial¶

    This step-by-step tutorial will guide you through making your first API calls to the Crew Appraisals Backend, from getting an access token to handling responses and errors.

    Prerequisites¶

    Before you start, ensure you have:

    • ✅ The API server running (see Quickstart Guide)
    • ✅ A user account (create one following the Authentication Guide)
    • ✅ Your preferred HTTP client (cURL, Python, JavaScript, or Postman)

    Tutorial Overview¶

    We'll walk through these steps:

    1. Register a new user (if you don't have one)
    2. Login to get an access token
    3. Make authenticated API calls
    4. Handle responses and errors
    5. Explore different endpoints

    Step 1: Verify API is Running¶

    First, let's check that the API is accessible:

    Step 6: Understanding Responses¶

    Success Responses¶

    The API uses standard HTTP status codes:

    Code Meaning Description
    200 OK Request successful
    201 Created Resource created successfully
    204 No Content Request successful, no content returned

    Error Responses¶

    Common error responses and their meanings:

    {
  "detail": "Could not validate credentials"
}

    Causes: - Missing Authorization header - Invalid or expired access token - Malformed token

    {
  "detail": "Not enough permissions"
}

    Causes: - User role doesn't have required permissions - Account is deactivated - Resource access denied

    {
  "detail": "User not found"
}

    Causes: - Endpoint doesn't exist - Resource not found - Invalid resource ID

    {
  "detail": [
    {
      "loc": ["body", "email"],
      "msg": "field required",
      "type": "value_error.missing"
    }
  ]
}

    Causes: - Missing required fields - Invalid data format - Validation rules not met

    Step 7: Error Handling Examples¶

    Here's how to handle different error scenarios:

    import requests
from requests.exceptions import RequestException, Timeout, ConnectionError

def make_authenticated_request(url, access_token, method='GET', **kwargs):
    """
    Make an authenticated request with proper error handling
    """
    headers = kwargs.get('headers', {})
    headers['Authorization'] = f'Bearer {access_token}'
    kwargs['headers'] = headers

    try:
        response = requests.request(method, url, timeout=30, **kwargs)

        # Handle different status codes
        if response.status_code == 200:
            return response.json(), None

        elif response.status_code == 401:
            return None, "Authentication failed - please login again"

        elif response.status_code == 403:
            return None, "Access denied - insufficient permissions"

        elif response.status_code == 404:
            return None, "Resource not found"

        elif response.status_code == 422:
            error_detail = response.json().get('detail', 'Validation error')
            return None, f"Validation error: {error_detail}"

        elif response.status_code >= 500:
            return None, "Server error - please try again later"

        else:
            return None, f"Request failed with status {response.status_code}"

    except Timeout:
        return None, "Request timed out - please try again"

    except ConnectionError:
        return None, "Connection error - check if the server is running"

    except RequestException as e:
        return None, f"Request error: {str(e)}"

# Example usage
data, error = make_authenticated_request(
    "http://localhost:8000/api/v1/auth/me",
    access_token
)

if data:
    print("Success:", data)
else:
    print("Error:", error)

    async function makeAuthenticatedRequest(url, accessToken, options = {}) {
  /**
   * Make an authenticated request with proper error handling
   */
  const headers = {
    'Authorization': `Bearer ${accessToken}`,
    ...options.headers
  };

  try {
    const response = await fetch(url, {
      timeout: 30000,
      ...options,
      headers
    });

    // Handle different status codes
    if (response.ok) {
      return { data: await response.json(), error: null };
    }

    switch (response.status) {
      case 401:
        return { data: null, error: 'Authentication failed - please login again' };

      case 403:
        return { data: null, error: 'Access denied - insufficient permissions' };

      case 404:
        return { data: null, error: 'Resource not found' };

      case 422:
        const errorDetail = await response.json();
        return {
          data: null,
          error: `Validation error: ${JSON.stringify(errorDetail.detail)}`
        };

      case 500:
      case 502:
      case 503:
        return { data: null, error: 'Server error - please try again later' };

      default:
        return { data: null, error: `Request failed with status ${response.status}` };
    }

  } catch (error) {
    if (error.name === 'TypeError' && error.message.includes('fetch')) {
      return { data: null, error: 'Connection error - check if the server is running' };
    }
    return { data: null, error: `Request error: ${error.message}` };
  }
}

// Example usage
const { data, error } = await makeAuthenticatedRequest(
  'http://localhost:8000/api/v1/auth/me',
  accessToken
);

if (data) {
  console.log('Success:', data);
} else {
  console.error('Error:', error);
}

    Complete Working Example¶

    Here's a complete example that puts everything together:

    #!/usr/bin/env python3
"""
Complete example of interacting with the Crew Appraisals API
"""

import requests
import json
import time
from datetime import datetime

class CrewAppraisalsClient:
    def __init__(self, base_url="http://localhost:8000"):
        self.base_url = base_url
        self.access_token = None
        self.refresh_token = None
        self.user_info = None

    def register(self, username, email, password, first_name, last_name, role="crew_member"):
        """Register a new user"""
        print(f"🔐 Registering user: {username}")

        response = requests.post(f"{self.base_url}/api/v1/auth/register", json={
            "username": username,
            "email": email,
            "password": password,
            "first_name": first_name,
            "last_name": last_name,
            "role": role
        })

        if response.status_code == 201:
            user = response.json()
            print(f"✅ User registered successfully: {user['id']}")
            return True
        elif response.status_code == 409:
            print("⚠️  User already exists")
            return False
        else:
            print(f"❌ Registration failed: {response.text}")
            return False

    def login(self, username, password):
        """Login and store tokens"""
        print(f"🔑 Logging in as: {username}")

        response = requests.post(f"{self.base_url}/api/v1/auth/login", json={
            "username": username,
            "password": password
        })

        if response.status_code == 200:
            auth_data = response.json()
            self.access_token = auth_data["access_token"]
            self.refresh_token = auth_data["refresh_token"]
            self.user_info = auth_data["user"]

            print(f"✅ Login successful!")
            print(f"   User: {self.user_info['username']} ({self.user_info['role']})")
            print(f"   Token expires in: {auth_data['expires_in']} seconds")
            return True
        else:
            print(f"❌ Login failed: {response.text}")
            return False

    def get_headers(self):
        """Get authorization headers"""
        if not self.access_token:
            raise Exception("Not authenticated - please login first")
        return {"Authorization": f"Bearer {self.access_token}"}

    def get_current_user(self):
        """Get current user info"""
        print("👤 Getting current user info...")

        response = requests.get(
            f"{self.base_url}/api/v1/auth/me",
            headers=self.get_headers()
        )

        if response.status_code == 200:
            user = response.json()
            print(f"✅ Current user: {user['username']}")
            return user
        else:
            print(f"❌ Failed to get user info: {response.text}")
            return None

    def get_session_info(self):
        """Get current session info"""
        print("🔍 Getting session info...")

        response = requests.get(
            f"{self.base_url}/api/v1/auth/session",
            headers=self.get_headers()
        )

        if response.status_code == 200:
            session = response.json()
            print(f"✅ Session ID: {session['id']}")
            print(f"   Created: {session['created_at']}")
            print(f"   Expires: {session['expires_at']}")
            print(f"   Last Activity: {session['last_activity']}")
            return session
        else:
            print(f"❌ Failed to get session info: {response.text}")
            return None

    def logout(self):
        """Logout and clear tokens"""
        print("🚪 Logging out...")

        if self.access_token:
            response = requests.post(
                f"{self.base_url}/api/v1/auth/logout",
                headers=self.get_headers()
            )

            if response.status_code == 200:
                print("✅ Logged out successfully")
            else:
                print(f"⚠️  Logout request failed: {response.text}")

        # Clear local tokens
        self.access_token = None
        self.refresh_token = None
        self.user_info = None

    def check_api_health(self):
        """Check API health"""
        print("🏥 Checking API health...")

        response = requests.get(f"{self.base_url}/health")

        if response.status_code == 200:
            health = response.json()
            print(f"✅ API is {health['status']}")
            print(f"   Database: {health['database']}")
            print(f"   Version: {health['version']}")
            return True
        else:
            print(f"❌ API health check failed: {response.status_code}")
            return False

def main():
    """Main example function"""
    print("🚀 Crew Appraisals API - First API Call Tutorial\n")

    # Initialize client
    client = CrewAppraisalsClient()

    # Check API health
    if not client.check_api_health():
        return

    print()

    # Register a user (or skip if exists)
    username = f"demo_user_{int(time.time())}"  # Unique username
    client.register(
        username=username,
        email=f"{username}@example.com",
        password="SecurePassword123!",
        first_name="Demo",
        last_name="User"
    )

    print()

    # Login
    if client.login(username, "SecurePassword123!"):
        print()

        # Make authenticated requests
        client.get_current_user()
        print()

        client.get_session_info()
        print()

        # Logout
        client.logout()

    print("\n🎉 Tutorial completed successfully!")
    print("\nNext steps:")
    print("- Explore the API documentation at: http://localhost:8000/docs")
    print("- Try different endpoints and methods")
    print("- Implement error handling and token refresh")

if __name__ == "__main__":
    main()

        return false;
      }
    } catch (error) {
      console.error('❌ Login error:', error);
      return false;
    }
  }

  getHeaders() {
    if (!this.accessToken) {
      throw new Error('Not authenticated - please login first');
    }
    return { 'Authorization': `Bearer ${this.accessToken}` };
  }

  async getCurrentUser() {
    console.log('👤 Getting current user info...');

    try {
      const response = await fetch(`${this.baseUrl}/api/v1/auth/me`, {
        headers: this.getHeaders()
      });

      if (response.ok) {
        const user = await response.json();
        console.log(`✅ Current user: ${user.username}`);
        return user;
      } else {
        const error = await response.text();
        console.log(`❌ Failed to get user info: ${error}`);
        return null;
      }
    } catch (error) {
      console.error('❌ Get user error:', error);
      return null;
    }
  }

  async getSessionInfo() {
    console.log('🔍 Getting session info...');

    try {
      const response = await fetch(`${this.baseUrl}/api/v1/auth/session`, {
        headers: this.getHeaders()
      });

      if (response.ok) {
        const session = await response.json();
        console.log(`✅ Session ID: ${session.id}`);
        console.log(`   Created: ${session.created_at}`);
        console.log(`   Expires: ${session.expires_at}`);
        console.log(`   Last Activity: ${session.last_activity}`);
        return session;
      } else {
        const error = await response.text();
        console.log(`❌ Failed to get session info: ${error}`);
        return null;
      }
    } catch (error) {
      console.error('❌ Get session error:', error);
      return null;
    }
  }

  async logout() {
    console.log('🚪 Logging out...');

    try {
      if (this.accessToken) {
        const response = await fetch(`${this.baseUrl}/api/v1/auth/logout`, {
          method: 'POST',
          headers: this.getHeaders()
        });

        if (response.ok) {
          console.log('✅ Logged out successfully');
        } else {
          const error = await response.text();
          console.log(`⚠️  Logout request failed: ${error}`);
        }
      }
    } catch (error) {
      console.error('❌ Logout error:', error);
    }

    // Clear local tokens
    this.accessToken = null;
    this.refreshToken = null;
    this.userInfo = null;
  }

  async checkApiHealth() {
    console.log('🏥 Checking API health...');

    try {
      const response = await fetch(`${this.baseUrl}/health`);

      if (response.ok) {
        const health = await response.json();
        console.log(`✅ API is ${health.status}`);
        console.log(`   Database: ${health.database}`);
        console.log(`   Version: ${health.version}`);
        return true;
      } else {
        console.log(`❌ API health check failed: ${response.status}`);
        return false;
      }
    } catch (error) {
      console.error('❌ Health check error:', error);
      return false;
    }
  }
}

async function main() {
  console.log('🚀 Crew Appraisals API - First API Call Tutorial\n');

  // Initialize client
  const client = new CrewAppraisalsClient();

  if (!(await client.checkApiHealth())) {
