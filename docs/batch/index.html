    Batch Processing Guide¶

    Overview¶

    NetIntel-OCR provides efficient batch processing capabilities for handling multiple documents with parallel processing, progress tracking, and centralized storage.

    Basic Batch Processing¶

    Process Multiple Files¶

    # Process all PDFs in directory
netintel-ocr --batch *.pdf

# With specific model
netintel-ocr --batch --model qwen2.5vl:7b *.pdf

# Recursive processing
netintel-ocr --batch --recursive /path/to/documents

    Parallel Processing¶

    # Process 4 documents simultaneously
netintel-ocr --batch --max-parallel 4 *.pdf

# Auto-detect optimal parallelism
netintel-ocr --batch --auto-parallel *.pdf

# GPU parallel processing
netintel-ocr --batch --gpu --max-parallel 2 *.pdf

    Advanced Batch Features¶

    Batch Ingestion¶

    # Ingest to vector store
netintel-ocr --batch-ingest \
  --collection network_docs \
  --max-parallel 4 \
  *.pdf

# With deduplication
netintel-ocr --batch-ingest \
  --deduplicate \
  --collection unified_docs \
  /path/to/documents/*.pdf

    Progress Tracking¶

    # Enable progress bar
netintel-ocr --batch --progress *.pdf

# Save progress for resume
netintel-ocr --batch \
  --checkpoint batch-checkpoint.json \
  --resume-on-failure \
  *.pdf

    Output Organization¶

    # Organize by document type
netintel-ocr --batch \
  --output-structure type \
  --output-dir processed/ \
  *.pdf

# Result:
# processed/
#   ├── network_diagrams/
#   ├── flow_diagrams/
#   └── text_only/

    Batch Configuration¶

    YAML Configuration¶

    # batch-config.yaml
batch:
  max_parallel: 4
  chunk_size: 10
  resume_on_failure: true
  checkpoint_file: batch-state.json

  output:
    structure: document  # or 'type', 'date'
    dir: ./processed

  models:
    text: Nanonets-OCR-s:latest
    network: qwen2.5vl:7b
    flow: qwen2.5vl:7b

  filters:
    min_pages: 5
    max_pages: 500
    file_types: [pdf, png, jpg]

  error_handling:
    max_retries: 3
    retry_delay: 5
    skip_on_error: false

    Use Configuration¶

    netintel-ocr --batch --config batch-config.yaml *.pdf

    Centralized Database¶

    Merge to Central Store¶

    # Create centralized database
netintel-ocr --merge-to-centralized \
  --source-dir ./processed \
  --central-db ./central/unified.db

# With metadata
netintel-ocr --merge-to-centralized \
  --add-metadata "project=network-refresh" \
  --add-metadata "date=2024-01-15" \
  ./processed/* ./central/unified.db

    Query Centralized Database¶

    # Search across all documents
netintel-ocr --query \
  --db ./central/unified.db \
  "firewall configuration"

# Filter by metadata
netintel-ocr --query \
  --db ./central/unified.db \
  --filter "project=network-refresh" \
  "DMZ architecture"

    Cloud Storage Integration¶

    S3/MinIO Support¶

    # Configure S3
export AWS_ACCESS_KEY_ID=your-key
export AWS_SECRET_ACCESS_KEY=your-secret
export S3_BUCKET=netintel-output

# Process and upload to S3
netintel-ocr --batch \
  --output-s3 s3://netintel-output/processed/ \
  *.pdf

# Process from S3
netintel-ocr --batch \
  --input-s3 s3://netintel-input/*.pdf \
  --output-s3 s3://netintel-output/

    Azure Blob Storage¶

    # Configure Azure
export AZURE_STORAGE_CONNECTION_STRING=your-connection-string

# Process with Azure storage
netintel-ocr --batch \
  --storage-backend azure \
  --container processed-docs \
  *.pdf

    Performance Optimization¶

    Memory Management¶

    # Limit memory per process
netintel-ocr --batch \
  --max-memory 4GB \
  --max-parallel 2 \
  *.pdf

# Enable swap for large documents
netintel-ocr --batch \
  --enable-swap \
  --swap-dir /tmp/netintel-swap \
  large-docs/*.pdf

    CPU/GPU Optimization¶

    # CPU-only batch processing
netintel-ocr --batch \
  --cpu-only \
  --max-parallel $(nproc) \
  *.pdf

# Mixed CPU/GPU processing
netintel-ocr --batch \
  --gpu-for-models "llava,qwen2.5vl" \
  --cpu-for-models "Nanonets-OCR-s" \
  *.pdf

    Caching Strategy¶

    # Enable aggressive caching
netintel-ocr --batch \
  --cache-models \
  --cache-embeddings \
  --cache-dir /tmp/netintel-cache \
  *.pdf

# Share cache across runs
export NETINTEL_CACHE_DIR=/shared/cache
netintel-ocr --batch --use-cache *.pdf

    Monitoring and Logging¶

    Real-time Monitoring¶

    # Enable metrics server
netintel-ocr --batch \
  --metrics-port 9090 \
  --progress-webhook http://monitor/progress \
  *.pdf

# View metrics
curl http://localhost:9090/metrics

    Detailed Logging¶

    # Per-document logs
netintel-ocr --batch \
  --log-level DEBUG \
  --log-per-document \
  --log-dir ./logs \
  *.pdf

# Structured logging
netintel-ocr --batch \
  --log-format json \
  --log-file batch.jsonl \
  *.pdf

    Error Handling¶

    Retry Logic¶

    # Automatic retry with backoff
netintel-ocr --batch \
  --max-retries 3 \
  --retry-backoff exponential \
  --retry-delay 5 \
  *.pdf

    Failed Document Handling¶

    # Skip failed documents
netintel-ocr --batch \
  --skip-on-error \
  --failed-list failed.txt \
  *.pdf

# Reprocess failed documents
netintel-ocr --batch \
  --input-list failed.txt \
  --max-retries 5

    Batch Scripts¶

    Shell Script Example¶

    #!/bin/bash
# batch-process.sh

DOCS_DIR="/path/to/documents"
OUTPUT_DIR="/path/to/output"
FAILED_LIST="failed_docs.txt"

# Clear previous failures
> $FAILED_LIST

# Process in chunks
find $DOCS_DIR -name "*.pdf" | while read -r file; do
  netintel-ocr \
    --model qwen2.5vl:7b \
    --output-dir $OUTPUT_DIR \
    "$file" || echo "$file" >> $FAILED_LIST
done

# Retry failed documents
if [ -s $FAILED_LIST ]; then
  echo "Retrying failed documents..."
  while read -r file; do
    netintel-ocr --model minicpm-v:latest "$file"
  done < $FAILED_LIST
fi

    Python Script Example¶

    # batch_processor.py
import os
from pathlib import Path
from netintel_ocr import BatchProcessor

processor = BatchProcessor(
    max_parallel=4,
    model="qwen2.5vl:7b",
    output_dir="./processed"
)

# Process all PDFs
pdf_files = Path("/documents").glob("**/*.pdf")
results = processor.process_batch(pdf_files)

# Handle results
for result in results:
    if result.success:
        print(f"✓ {result.file}: {result.diagrams_found} diagrams")
    else:
        print(f"✗ {result.file}: {result.error}")

# Generate summary
processor.generate_summary("batch_summary.json")

    Best Practices¶

    1. Chunk Large Batches: Process in groups of 50-100 documents
    2. Use Checkpoints: Enable resume for long-running batches
    3. Monitor Memory: Set limits to prevent OOM errors
    4. Deduplicate First: Remove duplicates before processing
    5. Test Small Sample: Validate settings on subset first
