    Customer Service Chat Intelligence¶

    Scenario Overview¶

    Organization: Global Telecom Service Provider with B2B focus
    Challenge: Enable intelligent customer service for complex enterprise offerings
    Scale: 10,000+ enterprise customers, 500+ service products, 200+ locations
    Integration: ServiceNow CMDB, Salesforce CRM, Product Literature

    Technical Workflow¶

    Step 1: Initialize Customer Service Intelligence¶

    # Setup customer service environment
mkdir -p customer-service/{literature,cmdb,pricing,analysis}
cd customer-service

# Configure multi-source integration
cat > service_config.yaml << EOF
collection: customer_service
update_interval: 3600  # Hourly updates
data_sources:
  - servicenow:
      endpoint: ${SERVICENOW_ENDPOINT}
      api_key: ${SERVICENOW_API_KEY}
      tables: [cmdb_ci, locations, service_offering]
  - salesforce:
      endpoint: ${SALESFORCE_ENDPOINT}
      api_key: ${SALESFORCE_API_KEY}
      objects: [Product2, PricebookEntry, ServiceContract]
  - literature:
      path: literature/
      types: [pdf, docx, pptx]
kg_model: RotatE
embedding_model: qwen3-embedding:8b
llm_model: gemma3:4b-it-qat
service_categories:
  - SDWAN
  - Firewall_as_Service
  - Zero_Trust_VPN
  - SASE
  - 5G_Private_Network
  - Edge_Computing
EOF

# Initialize with customer service focus
netintel-ocr kg init --config service_config.yaml --mode customer-service

    Step 2: Ingest Service Documentation¶

    # Process solution literature
netintel-ocr --batch \
  --kg-model RotatE \
  --extract-service-features \
  --extract-pricing-tiers \
  --extract-technical-specs \
  literature/solutions/*.pdf \
  --output analysis/solutions/

# Process service datasheets
for datasheet in literature/datasheets/*.pdf; do
  netintel-ocr \
    --extract-tables \
    --extract-diagrams \
    --kg-store-offerings \
    "$datasheet" \
    --output "analysis/datasheets/$(basename $datasheet .pdf)/"
done

# Build service knowledge graph
netintel-ocr kg build-service-catalog \
  --collection customer_service \
  --include-features \
  --include-pricing \
  --include-locations \
  --output analysis/service_catalog.json

    Integration APIs¶

    REST API for Chat Systems¶

    from netintel_ocr.customer_service import ServiceBot
import os

# Initialize service bot
bot = ServiceBot(
    collection="customer_service",
    ollama_host=os.environ.get('OLLAMA_HOST'),
    servicenow_endpoint=os.environ.get('SERVICENOW_ENDPOINT'),
    salesforce_endpoint=os.environ.get('SALESFORCE_ENDPOINT')
)

# Process customer query
async def handle_chat_query(customer_id: str, query: str, context: dict):
    # Get customer profile from CRM
    customer = await bot.get_customer_profile(customer_id)

    # Find relevant services
    services = await bot.match_services(
        query=query,
        customer_profile=customer,
        include_pricing=True,
        include_availability=True
    )

    # Generate response
    response = await bot.generate_response(
        services=services,
        style="consultative",
        include_next_steps=True
    )

    return response

# Webhook for live chat integration
@app.post("/chat/webhook")
async def chat_webhook(request: ChatRequest):
    response = await handle_chat_query(
        customer_id=request.customer_id,
        query=request.message,
        context=request.context
    )

    # Log interaction for analytics
    await bot.log_interaction(
        customer_id=request.customer_id,
        query=request.message,
        response=response,
        channel="web_chat"
    )

    return {"response": response, "confidence": response.confidence}

    Salesforce Integration¶

    # Sync with Salesforce for real-time pricing
from netintel_ocr.integrations import SalesforceSync

sync = SalesforceSync(
    collection="customer_service",
    sf_endpoint=os.environ.get('SALESFORCE_ENDPOINT'),
    sf_token=os.environ.get('SALESFORCE_TOKEN')
)

# Update pricing from Salesforce
async def update_pricing():
    # Pull latest pricing
    pricing = await sync.get_current_pricing(
        products=["SDWAN", "FWaaS", "Zero_Trust_VPN"],
        include_promotions=True
    )

    # Update knowledge graph
    await sync.update_kg_pricing(pricing)

    # Refresh embeddings for similarity search
    await sync.refresh_pricing_embeddings()

# Schedule hourly updates
schedule.every().hour.do(update_pricing)
