    Network Infrastructure Migration Planning¶

    Scenario Overview¶

    Organization: Telecommunications provider with 50 data centers
    Challenge: Migrate to cloud infrastructure with zero unplanned downtime
    Documentation: 523 network PDFs, 1,200+ configuration files
    Complexity: Unknown dependencies between legacy and modern systems

    Technical Workflow¶

    Step 1: Environment Setup¶

    # Install NetIntel-OCR with KG support (included by default in v0.1.17)
pip install netintel-ocr==0.1.17

# Configure external Ollama server
export OLLAMA_HOST="http://your-ollama-server:11434"

# Verify Ollama has required models
curl $OLLAMA_HOST/api/tags | jq '.models[].name'
# Required: qwen2.5vl:7b, gemma3:4b-it-qat, qwen3-embedding:8b

# Start required graph/vector services
docker-compose up -d falkordb milvus

# Verify services
netintel-ocr kg check-requirements

# Expected output:
# ✅ FalkorDB: Connected (localhost:6379)
# ✅ Milvus: Connected (localhost:19530)
# ✅ Ollama: Connected (http://your-ollama-server:11434)
# ✅ PyKEEN: Installed (8 models available)
# ✅ Required Models: Available

    Step 2: Process Infrastructure Documentation¶

    # Create project directory
mkdir -p telecom-migration/{input,output,analysis}
cd telecom-migration

# Copy all documentation to input directory
cp /shared/docs/network/*.pdf input/
cp /shared/docs/configs/*.pdf input/
cp /shared/docs/architecture/*.pdf input/

# Process all PDFs with Knowledge Graph extraction
netintel-ocr --batch \
  --kg-model RotatE \
  --kg-epochs 200 \
  --kg-batch-size 512 \
  --max-parallel 8 \
  --output output/ \
  --collection telecom_migration \
  input/*.pdf

# Monitor processing progress
tail -f output/processing.log

# Expected output:
# [2024-01-15 10:23:45] Processing: network-datacenter-east.pdf
# [2024-01-15 10:23:47] Detected network diagram on page 3
# [2024-01-15 10:23:48] Extracted 47 entities, 156 relationships
# [2024-01-15 10:23:50] Stored in FalkorDB graph
# ...
# [2024-01-15 11:45:23] Completed: 523 documents processed
# [2024-01-15 11:45:23] Total entities: 3,847
# [2024-01-15 11:45:23] Total relationships: 12,493

    Step 3: Train Knowledge Graph Embeddings¶

    # Train KG embeddings on extracted graph
netintel-ocr kg train \
  --collection telecom_migration \
  --model RotatE \
  --epochs 200 \
  --embedding-dim 200 \
  --batch-size 512 \
  --gpu cuda:0 \
  --save-path output/kg_embeddings

# Monitor training metrics
tensorboard --logdir output/kg_embeddings/logs

# Verify embeddings
netintel-ocr kg stats --collection telecom_migration

# Output:
# Knowledge Graph Statistics:
#   Entities: 3,847
#   Relationships: 12,493
#   Entity Types: 15
#   Relation Types: 23
#   Embeddings: 200D vectors
#   Model Performance:
#     MRR: 0.89
#     Hits@10: 0.94
#     Training Loss: 0.0234

    Step 4: Dependency Analysis¶

    # Find all dependencies of critical systems
netintel-ocr kg query \
  --collection telecom_migration \
  --type dependencies \
  --entity "BillingSystem-PROD" \
  --max-depth 3 \
  --output-format json > analysis/billing_dependencies.json

# Analyze the dependencies
cat analysis/billing_dependencies.json | jq '.dependencies[] | select(.criticality > 8)'

# Output:
# {
#   "name": "Oracle-RAC-01",
#   "type": "database",
#   "criticality": 10,
#   "path": ["BillingSystem-PROD", "DB-Cluster", "Oracle-RAC-01"],
#   "migration_impact": "high",
#   "downtime_required": "4 hours"
# }

# Find all single points of failure
netintel-ocr kg analyze \
  --collection telecom_migration \
  --analysis-type single-points-of-failure \
  --output analysis/spof.json

# Generate migration waves based on dependencies
netintel-ocr kg migration-plan \
  --collection telecom_migration \
  --target-architecture cloud-native \
  --max-downtime 4h \
  --output analysis/migration_waves.json

# View migration waves
cat analysis/migration_waves.json | jq '.waves[] | {number: .wave_number, components: .components | length, risk: .risk_level}'

# Output:
# {"number": 1, "components": 23, "risk": "low"}
# {"number": 2, "components": 45, "risk": "low"}
# {"number": 3, "components": 67, "risk": "medium"}
# {"number": 4, "components": 89, "risk": "medium"}
# {"number": 5, "components": 112, "risk": "high"}
# {"number": 6, "components": 34, "risk": "critical"}

    Step 5: Cloud Compatibility Check¶

    # Check cloud compatibility for all components
netintel-ocr kg cloud-check \
  --collection telecom_migration \
  --provider aws \
  --region us-east-1 \
  --output analysis/cloud_compatibility.json

# Find incompatible configurations
cat analysis/cloud_compatibility.json | jq '.incompatible[] | {component: .name, reason: .incompatibility_reason}'

# Generate configuration migration scripts
netintel-ocr kg generate-configs \
  --collection telecom_migration \
  --source-format cisco-ios \
  --target-format aws-vpc \
  --output analysis/config_scripts/

# List generated scripts
ls -la analysis/config_scripts/
# vpc_setup.tf
# security_groups.tf
# route_tables.tf
# network_acls.tf

    Step 6: Generate Migration Report¶

    # Generate comprehensive migration report
netintel-ocr kg report \
  --collection telecom_migration \
  --report-type migration \
  --include-dependencies \
  --include-risks \
  --include-timeline \
  --format pdf \
  --output analysis/migration_report.pdf

# Generate executive summary
netintel-ocr kg report \
  --collection telecom_migration \
  --report-type executive-summary \
  --format markdown \
  --output analysis/executive_summary.md

# View key metrics
grep -E "Dependencies Found|Hidden Connections|Migration Waves|Estimated Duration|Risk Score" analysis/executive_summary.md

# Output:
# Dependencies Found: 12,493
# Hidden Connections: 23 (undocumented)
# Migration Waves: 6
# Estimated Duration: 8 weeks
# Risk Score: Medium (6.2/10)

    Validation and Testing¶

    # Validate migration plan completeness
netintel-ocr kg validate \
  --collection telecom_migration \
  --validation-type migration-completeness

# Output:
# ✅ All components included in migration plan
# ✅ No circular dependencies detected
# ✅ All critical paths identified
# ✅ Rollback procedures defined
# ⚠️  3 components need manual verification

# Compare with manual analysis (if available)
netintel-ocr kg compare \
  --collection telecom_migration \
  --manual-analysis /shared/docs/manual_dependencies.xlsx \
  --output analysis/comparison.json

# Results:
# Accuracy: 94% (vs 67% manual)
# Additional findings: 23 connections not in manual analysis
# Time saved: 3 months → 2 days

    Advanced Graph Queries¶

    Finding Hidden Dependencies¶

    # Use Cypher to find undocumented connections
netintel-ocr kg query \
  --cypher "MATCH p=(a:Component)-[*2..4]-(b:Component) 
            WHERE NOT (a)-[:DOCUMENTED_CONNECTION]-(b) 
            AND a.criticality > 7 
            AND b.criticality > 7 
            RETURN p LIMIT 20" \
  --output analysis/hidden_paths.json

    Impact Analysis¶

    # What happens if this component fails?
netintel-ocr kg impact \
  --component "Core-Router-DC1" \
  --cascade-depth 5 \
  --include-services \
  --include-customers \
  --output analysis/router_impact.json

# View affected services
cat analysis/router_impact.json | jq '.affected_services[] | {name: .service, customers: .customer_count, revenue_impact: .hourly_revenue}'

    Migration Risk Assessment¶

    # Calculate risk score for each wave
for i in {1..6}; do
  netintel-ocr kg risk-score \
    --collection telecom_migration \
    --wave $i \
    --factors "complexity,dependencies,criticality,testing" \
    --output "analysis/wave_${i}_risk.json"
done

# Aggregate risk report
netintel-ocr kg aggregate-risk \
  --risk-files "analysis/wave_*_risk.json" \
  --output analysis/total_migration_risk.json
